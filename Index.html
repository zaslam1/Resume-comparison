<!DOCTYPE html>

<html>

<head>

    <title>Job Description and Resume Comparison</title>

    <script>

        async function compare() {

            const jobDescription = document.getElementById('job_description').value;

            const resume = document.getElementById('resume').value;


            const response = await fetch('https://6381-34-125-27-149.ngrok-free.app/compare', {

                method: 'POST',

                headers: {

                    'Content-Type': 'application/json'

                },

                body: JSON.stringify({ job_description: jobDescription, resume: resume })

            });


            const result = await response.json();

            document.getElementById('result').innerText = result.result;

        }

    </script>

</head>

<body>

    <h1>Job Description and Resume Comparison</h1>

    <label for="job_description">Job Description:</label><br>

    <textarea id="job_description" rows="4" cols="50"></textarea><br><br>

    <label for="resume">Resume:</label><br>

    <textarea id="resume" rows="4" cols="50"></textarea><br><br>

    <button onclick="compare()">Compare</button>

    <h2>Result:</h2>

    <pre id="result"></pre>

</body>
